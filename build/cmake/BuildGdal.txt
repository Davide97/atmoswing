cmake_minimum_required(VERSION 3.0)

project(libgdal)

include(ExternalProject)

if (MINGW OR MSYS OR UNIX)
    ExternalProject_Add(libgdal
            DEPENDS curl jasper
            URL http://download.osgeo.org/gdal/CURRENT/gdal-${GDAL_VERSION}.tar.gz
            BUILD_IN_SOURCE 1
            CONFIGURE_COMMAND ./configure --prefix=${EXTERNAL_DIR} --with-jasper=${EXTERNAL_DIR} --with-curl=${CURL_ROOT}
            BUILD_COMMAND make -j$(nproc)
            INSTALL_COMMAND make install
            )
elseif (WIN32)


endif ()
